哈夫曼树编码

import java.util.PriorityQueue; import java.util.Scanner; public class Main { // 哈夫曼树节点定义 static class Node implements Comparable { int weight; // 频次（权重） int id; // 排序ID：叶子节点为ASCII码，内部节点为自增ID(>=256) Node left; Node right; public Node(int weight, int id) { this.weight = weight; this.id = id; } // 核心排序规则：首先比较权重；权重相同时比较ID @Override public int compareTo(Node o) { if (this.weight != o.weight) { return Integer.compare(this.weight, o.weight); } return Integer.compare(this.id, o.id); } } public static void main(String[] args) { Scanner scanner = new Scanner(System.in); if (!scanner.hasNextInt()) { return; } int n = scanner.nextInt(); for (int i = 0; i < n; i++) { String s = scanner.next(); processHuffman(s); } scanner.close(); } private static void processHuffman(String s) { if (s == null || s.length() == 0) { System.out.println(""); return; } // 1. 统计字符频次 int[] freq = new int[256]; for (int i = 0; i < s.length(); i++) { freq[s.charAt(i)]++; } // 2. 将所有出现的字符放入优先队列 PriorityQueue pq = new PriorityQueue<>(); for (int i = 0; i < 256; i++) { if (freq[i] > 0) { // 叶子节点的ID直接设为它的ASCII码 pq.add(new Node(freq[i], i)); } } // 特殊情况处理：若只有一个独立字符 if (pq.size() == 1) { StringBuilder sb = new StringBuilder(); for (int i = 0; i < s.length(); i++) { sb.append("0"); } System.out.println(sb.toString()); return; } // 3. 循环合并节点构建哈夫曼树 int internalId = 256; // 内部节点ID从256开始自增 while (pq.size() > 1) { Node left = pq.poll(); Node right = pq.poll(); // 创建父节点 Node parent = new Node(left.weight + right.weight, internalId++); parent.left = left; parent.right = right; pq.add(parent); } // 4. 树根节点 Node root = pq.poll(); // 5. 递归生成哈夫曼编码字典 String[] codes = new String[256]; generateCodes(root, "", codes); // 6. 按照原字符串顺序输出编码结果 StringBuilder result = new StringBuilder(); for (int i = 0; i < s.length(); i++) { result.append(codes[s.charAt(i)]); } System.out.println(result.toString()); } // 深度优先遍历生成编码 private static void generateCodes(Node node, String currentCode, String[] codes) { if (node == null) { return; } // 检测是否为叶子节点 (叶子节点的ID < 256) if (node.left == null && node.right == null) { codes[node.id] = currentCode; return; } // 左往 0，右往 1 generateCodes(node.left, currentCode + "0", codes); generateCodes(node.right, currentCode + "1", codes); } }
